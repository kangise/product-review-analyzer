```json
{
  "usage_scenarios": [
    {
      "scenario_name": "场景名称",
      "frequency": "使用频率",
      "context": "使用情境描述",
      "user_goals": ["目标1", "目标2"],
      "pain_points": ["痛点1", "痛点2"],
      "success_factors": ["成功因素1", "成功因素2"]
    }
  ],
  "scenario_insights": ["洞察1", "洞察2"]
}
```
